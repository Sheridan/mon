function(add_daemon target_name)
  add_dependencies(${target_name} base)
  target_link_libraries(${target_name} base ${GLOBAL_LINK_LIBS})
  install(TARGETS ${target_name} RUNTIME DESTINATION bin )
endfunction(add_daemon)
